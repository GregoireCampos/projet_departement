# On veut crer une matrice avec les lignes : les diffrents critres et les colonnes : les diffrentes annes
M=[]
criteria = []
i = 0
while i < 13 :
    # on ajoute la ligne en cours dans la liste
    ligne = []
    curr_row = i
    row = worksheet.row(curr_row)
    curr_cell = 0
    while curr_cell < num_cells:
        cell_value = worksheet.cell_value(curr_row, curr_cell)
        if type(cell_value) == float :
            ligne.append(round(cell_value,4))
        else :
            ligne.append(cell_value)
        curr_cell += 1
    if i>0:
        criteria.append(str(ligne[0]))
    M.append(ligne[1:])
    i += 1
